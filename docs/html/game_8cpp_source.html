87 bool CGame::OnUserUpdate(float deltaTime)
88 {
89  try
90  {
92  //close game
93  if (GetKey(olc::ESCAPE).bHeld)
94  {
95  std::cout << "Bleep terminated with escape." << std::endl;
96  return false;
97  }
98  //move the world relatively to the player
99  if (GetKey(olc::W).bHeld)
100  m_velocity.y += m_acceleration.y * deltaTime;
101  if (GetKey(olc::S).bHeld)
102  m_velocity.y -= m_acceleration.y * deltaTime;
103  if (GetKey(olc::A).bHeld)
104  m_velocity.x += m_acceleration.x * deltaTime;
105  if (GetKey(olc::D).bHeld)
106  m_velocity.x -= m_acceleration.x * deltaTime;
107  if (m_velocity.mag2() > m_maxSpeed2)
108  m_velocity = m_velocity.norm() * m_maxSpeed;
109  //mouse input
110  auto pPlayer{ player() };
111  if (GetMouse(static_cast<int>(mouse::LEFT)).bHeld && pPlayer)
112  pPlayer->fireHawking(getCursor(), deltaTime);
113 
115  spawnDebris(deltaTime);
116 
117  //olc - needs to be called before drawing
118  Clear(olc::BLACK);
119  SetPixelMode(olc::Pixel::MASK);
120 
121  //User interface
122  drawGUI(deltaTime);
123 
124  //objects update & clean up
125  for (auto& objectType : m_objects)
126  for (auto objItr = objectType.begin(); objItr != objectType.end(); ++objItr)
127  {
128  if ((*objItr)->state() == objectStates::ALIVE)
129  (*objItr)->update(deltaTime);
130  else
131  objectType.erase(objItr++);
132 
133  if (objItr == objectType.end())
134  break;
135  }
136 
138  globalEffects(deltaTime);
139 
140  //olc - needs to be called after drawing
141  SetPixelMode(olc::Pixel::NORMAL);
142  }
143  catch (CException& exception)
144  {
145  exception.print();
146  return false;
147  }
148  catch (...)
149  {
150  std::cerr << "Unhandled exception in " << S(__FILE__);
151  return false;
152  }
153 
154  return true;
155 }

191 void CGame::initStars()
192 {
193  for (size_t i{ 0 }; i < m_starCardinality; ++i)
194  {
195  v2d starPos(static_cast<float>(rand() % ScreenWidth()), static_cast<float>(rand() % ScreenHeight()));
196 
197  colors starColor;
198  if (i > m_starCardinality * 0.9f)
199  starColor = olc::WHITE;
200  else if (i > m_starCardinality * 0.7f)
201  starColor = olc::GREY;
202  else if (i > m_starCardinality * 0.4f)
203  starColor = olc::DARK_GREY;
204  else
205  starColor = olc::VERY_DARK_GREY;
206 
207  std::shared_ptr<CStar> star{ std::make_shared<CStar>(this, objectTypes::STAR, starPos, 0, starColor) };
208  if ((center() - starPos).mag() <= radiusView())
209  {
210  star->kinetics()->speedFactor((1.5f - static_cast<float>(std::log10(m_starCardinality - i + 1) / std::log10(m_starCardinality + 1))) * 0.5f);
211  addObject(std::move(star));
212  }
213  }
214 }
